Demo

#1: Basic

<div id="demo1"> <img src="img/01.jpg" width="100"/> <img src="img/02.jpg" width="100"/> <img src="img/03.jpg" width="100"/> </div> <script type="text/javascript"> $('#demo1 img').rollover(); </script>

#2: initial selected image & no mouse leave eventt

<div id="demo2"> <img src="img/01.jpg" width="100"/> <img src="img/02.jpg" width="100" class="selected"/> <img src="img/03.jpg" width="100"/> </div> <script type="text/javascript"> $('#demo2 img').rollover({ out:false }); </script>

#3: set default image (auto select initial image when leave from container)

<div id="demo3"> <img src="img/01.jpg" width="100"/> <img src="img/02.jpg" width="100"/> <img src="img/03.jpg" width="100" class="default"/> </div> <script type="text/javascript"> $('#demo3 img').rollover({ out:false, container: $('#demo3') }); </script>

#4: layer feature

<div id="demo4" style="position:relative; height:120px;"> <img src="img/01.jpg" width="100" class="default"/> <img src="img/02.jpg" width="100"/> <img src="img/03.jpg" width="100"/> <div class="desc">image no.1</div> <div class="desc">image no.2</div> <div class="desc">image no.3</div> </div> <script type="text/javascript"> $('#demo4 img').rollover({ out:false, layers: $('#demo4 .desc'), container: $('#demo4'), layerOn: function(img,layer,index) { layer.css({position:'absolute', opacity:0, display:'block', top: 90, left: img.position().left}).animate({top:100, opacity:1}); }, layerOff: function(img,layer,index) { layer.animate({top:90,opacity:0},'fast'); } }); </script>

#5: pick over image path from class

<div id="demo5"> <img src="img/04_off.jpg" width="100" class="selected {over:'img/04_on.jpg'}"/> <img src="img/05_off.jpg" width="100" class="{over:'img/05_on.jpg'}"/> </div> <script type="text/javascript"> $('#demo5 img').rollover({ out:false, getOver: 'fromClass' }); </script>

#6: pick over image path as replace

<div id="demo6"> <img src="img/04_off.jpg" width="100"/> <img src="img/05_off.jpg" width="100"/> </div> <script type="text/javascript"> $('#demo6 img').rollover({ getOver: 'replaceOn', replaceOn: '_on', replaceOff: '_off' }); </script>
